  document.getElementById('v34-input').style.display = 'none';
  document.getElementById('v34-loading').style.display = 'block';

  let p = document.getElementById('v34-bar');
  let s = document.getElementById('v34-status');

  let int = setInterval(() => {
    t += 6.66; // Reaches 100% in exactly 15 seconds
    p.style.width = t + '%';

    if(t > 20 && t <= 50) {
      s.innerText = "Evaluating Corporate Leadership Portfolios...";
    } else if(t > 50 && t <= 80) {
      s.innerText = "Securing Fast-Track Institutional Waivers...";
    } else if(t > 80 && t < 100) {
      s.innerText = "Authorizing Elite Visa Compliance Modules...";
    }

    if(t >= 100) {
      clearInterval(int);
      setTimeout(() => {
        document.getElementById('v34-loading').style.display = 'none';
        document.getElementById('v34-result').style.display = 'block';
      }, 300);
    }
  }, 1000);
}
